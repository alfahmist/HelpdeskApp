CREATE PROCEDURE [dbo].[SP_Register]

	@Name varchar(200),
	@Email varchar(200),
	@BirthDate Date,
	@Gender varchar(50),
	@Password varchar(50),
	@RoleId as int,
	@DepartmentId as int,
	@Phone varchar(50)

AS
IF NOT EXISTS (SELECT * FROM TB_M_Employee)
BEGIN
	ALTER SEQUENCE SequenceEmployee RESTART WITH 1
END
BEGIN
	DECLARE @Nid varchar(50) = 'EMP' + FORMAT(NEXT VALUE FOR SequenceEmployee,'00')

	INSERT INTO TB_M_Employee([ID], [Name],Email, BirthDate, Gender, DepartmentId, Phone, RoleId) 
	VALUES (@Nid, @Name, @Email, @Birthdate, @Gender, @DepartmentId,@Phone, @RoleId)

	INSERT INTO TB_M_Account VALUES (@Nid, @Password)
END
